003import edu.wpi.first.wpilibj.command.Command;
004import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
005import edu.wpi.first.wpilibj.templates.OI;
006import edu.wpi.first.wpilibj.templates.subsystems.*;
007import edu.wpi.first.wpilibj.DriverStationLCD;
008
009import edu.gappleto.common.*;
010
011/**
012 * The base for all commands. All atomic commands should subclass CommandBase.
013 * CommandBase stores creates and stores each control system. To access a
014 * subsystem elsewhere in your code in your code use
015 * CommandBase.exampleSubsystem
016 *
017 * @author Gabe Appleton (gappleto) & Justin Hanna
018 */
019public abstract class CommandBase extends Command
020{
021    public static Cons Console = new Cons(DriverStationLCD.getInstance());
022    public static DriveTrain DriveTrain = new DriveTrain();
023    public static Launcher Jaws = new Launcher();
024    public static Pneumatics Air = new Pneumatics();
025    public static LED LED = new LED();
026    public static Logger Log = new Logger();
027    // Note: NEVER put anything that even POTENTIALLY uses the OI before the OI.
028    // It WILL crash the code.
029    public static OI oi;
030    // Create a single static instance of all of your subsystems
031
032    public static void init()
033    {
034        // This MUST be here. If the OI creates Commands (which it very likely
035        // will), constructing it during the construction of CommandBase (from
036        // which commands extend), subsystems are not guaranteed to be
037        // yet. Thus, their requires() statements may grab null pointers. Bad
038        // news. Don't move it.
039        Log.println("CommandBase initializing.");
040        oi = new OI();
041        Log.println("Operator Interface online.");
042        LED.setEnabled();
043        LED.setTeam();
044
045        // Show what command your subsystem is running on the SmartDashboard
046        SmartDashboard.putData(DriveTrain);
047        SmartDashboard.putData(Jaws);
048        SmartDashboard.putData(Air);
049    }
050
051    public static void fromAuton()
052    {
053        Air.fromAuton();
054    }
055
056    public static void enabled()
057    {
058        Log.println("Enabled.");
059        Air.InitShift();
060    }
061
062    public CommandBase(String name)
063    {
064        super(name);
065        Log.println("CommandBase initialized.");
066    }
067
068    public CommandBase()
069    {
070        super();
071        Log.println("CommandBase initialized.");
072    }
073}
